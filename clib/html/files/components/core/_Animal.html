(Animal has
  (superclasses (Organism))
  (required-slot ())
  (primary-slot ())
  (secondary-slot ())
  (description ("a living entity capable of locomotion"))
)

(every Animal has
  (capability ((a Agent-Role with
                  (in-event ((a Locomotion)))
  )))
)

(Animal has 
  (test-case ('(
                (*ShinerTheCat has (instance-of (Animal)))
                ((the instance-of of
                      (the in-event of
                           (the capability of *ShinerTheCat)))
                 = Locomotion
                )
))))
