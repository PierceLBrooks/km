	       (superclasses (Slot)))

;; Interface-Slot on load

#|
(every KM-Slot-Group has
               (domain (Thing))
               #|(range (Thing))|#
               (inverse ()) ; no inverse for this "grouper"
               (cardinality (N-to-N))
               (fluent-status (*Non-Fluent)))
|#
