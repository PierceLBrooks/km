(Room has
  (superclasses (Architectural-Structure))
  (description ("an architectural structure playing a container within a building"))
)

(every Room has
  (plays ((a Container)))
  (has-part ((a Floor) (a Wall) (a Ceiling)))
  (is-part-of ((a Building)))
)
