  (superclasses  (Increase))
#|(sentence-type (*punctual *conclusive))|#

  (instance-of (Increase))
  (from-value ((a Density-Value)))
)

(every Condense has
  (from-value ((a Density-Value with (density-of ((the base of Self))))))
  (base ((a Tangible-Entity with (density ((the from-value of Self))))))
)
