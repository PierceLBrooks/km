(Action has
	(superclasses 	     (Event))
	(primary-slot 	     (agent object instrument))
	(secondary-slot      (subevent))   
	(cmap-correspondence (
   		   	      (:seq object "the thing Acted upon")
   		              (:seq agent "the Actor")
 		              (:seq instrument "the thing used to Act")
	   		     )))
